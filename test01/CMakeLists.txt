cmake_minimum_required(VERSION 3.14 FATAL_ERROR)
project (Test VERSION 1.2.3.4)
# if() 条件专题 + message(FATAL_ERROR 报错中断)

# ON  true / OFF false
message("")
set(v0 ON) 
if(v0)
  message(STATUS "v0 true")
else()
  message(FATAL_ERROR "error")
endif()
message("")

# EQUAL 相等
set(idx -1)
if(idx EQUAL -1)
  message(STATUS "idx == -1 is true")
else()
  message(FATAL_ERROR "error")
endif()

set (A 0.1.2)
set (B 0.1.2)
if(A EQUAL B)
  message(STATUS "A == B is true")
else()
  message(FATAL_ERROR "error")
endif()
message("")

# if(TRUE)
set (A_B TRUE)
set (pB B)
if(A_${pB})
  message(STATUS "A_\${pB} is TRUE")
else()
  message(FATAL_ERROR "error")
endif()
message("")

# if(OFF)
set (str "OFF")
if(str)
  message(STATUS "str true")
else()
  message(FATAL_ERROR "jzz error")
endif()
